// 분수의 덧셈
// 슈도코드
/* 
    1) 새로운 분자를 계산한다
    numer = numer1 * denom2 + numer2 * denom1

    2) 새로운 분모를 계산한다
    denom = denom1 * denom2

    3) 분자와 분모의 최대 공약수를 구한다
    gcd = 최대공약수(numer, denom)

    4) 분자와 분모를 최대 공약수로 나눈다
    numer = numer / gcd
    denom = denom / gcd

    5) 배열에 담아서 반환한다 > [분자, 분모] 배열
    return [numer, denom]
*/
class Solution {
    public int[] solution(int numer1, int denom1, int numer2, int denom2) {
        
        int numer = numer1 * denom2 + numer2 * denom1;
        int denom = denom1 * denom2;

        int gcd = 1;

        // 최대공약수를 구하는 로직
        for(int i=1; i<=numer && i <= denom; i++){  // 두 수보다 적게 반복문 돌림
            if(numer % i == 0 && denom % i == 0) {  // 약수
                gcd = i;
            }
        }

        numer = numer / gcd;
        denom = denom / gcd;

        return new int[] {numer, denom};
    }
}



// 배열 두 배 만들기
// 슈도코드
/*
    1) numbers와 같은 크기의 answer 배열 생성
    2) numbers를 처음부터 끝까지 반복
    3) 현재 원소에 2를 곱한다
    4) answer의 같은 위치에 저장
    5) 반복이 끝나면 answer를 반환
*/
class Solution {
    public int[] solution(int[] numbers) {
        int[] answer = new int[numbers.length];

        for(int i=0; i<numbers.length; i++){
            answer[i] = numbers[i] * 2;
        }

        return answer;
    }
}
