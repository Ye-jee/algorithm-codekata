// 숫자 비교 1 - if문 사용
class Solution {
    public int solution(int num1, int num2) {
        
        if (num1 == num2) {
            return 1;
        }

        return -1;
    }
}

// 숫자 비교 2 - 삼항연산자 사용
class Solution {
    public int solution(int num1, int num2) {
        
        int answer = num1 == num2 ? 1 : -1;
        return answer;
    }
}
