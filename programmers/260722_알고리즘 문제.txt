package programmers;

// 콜라츠 추측
class Solution {
    public int solution(int num) {

        // 홀수일 때 3을 곱하는 과정에서 int 범위를 넘을 수 있으므로
        // 계산용 값은 long 타입으로 저장
        long value = num;

        // 콜라츠 연산을 수행한 횟수
        int count = 0;

        // 숫자가 1이 될 때까지 반복하되, 최대 연산 횟수는 500번으로 제한
        while (value != 1 && count < 500) {

            // 현재 값이 짝수라면 2로 나눔
            if (value % 2 == 0) {
                value /= 2;
            } else {
                // 현재 값이 홀수라면 3을 곱하고 1을 더함
                value = value * 3 + 1;
            }

            // 연산을 한 번 수행했으므로 횟수를 증가
            count++;
        }

        // 반복이 끝난 이유가 value가 1이 되었기 때문이라면
        // 지금까지 수행한 횟수를 반환
        if (value == 1) {
            return count;
        }

        // 500번을 수행해도 1이 되지 않았다면 -1을 반환
        return -1;
    }
}
